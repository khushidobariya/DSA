#include <stdio.h>
#include <conio.h>
int main(){
    clrscr();
    int i,j,k,matrix1[50][50],matrix2[50][50],prod[50][50],rsize,csize,rsize2,csize2;
    printf("Enter the number of rows in the first matrix: ");
    scanf("%d",&rsize);
    printf("Enter the number of columns in the first matrix: ");
    scanf("%d",&csize);
    printf("Enter the number of rows in the second matrix: ");
    scanf("%d",&rsize2);
    printf("Enter the number of columns in the second matrix: ");
    scanf("%d",&csize2);
    if(csize==rsize2){
        printf("Enter the elements of first matrix:\n");
        for(i=0;i<rsize;i++){
            for(j=0;j<csize;j++){
                printf("Enter element at row %d and column %d: ",i+1,j+1);
                scanf("%d",&matrix1[i][j]);
            }
        }
        printf("Enter the elements of second matrix:\n");
        for(i=0;i<rsize2;i++){
            for(j=0;j<csize2;j++){
                printf("Enter element at row %d and column %d: ",i+1,j+1);
                scanf("%d",&matrix2[i][j]);
            }
        }
         for(i=0;i<rsize;i++){
            for(j=0; j<csize2;j++){
                prod[i][j]=0;
            }
        }
        for(i=0;i<rsize;i++){
            for(j=0;j<csize2;j++){
                for(k=0;k<csize;k++) {
                    prod[i][j]+=matrix1[i][k]*matrix2[k][j];
                }
            }
        }
        printf("Your product matrix is:\n");
        for(i=0;i<rsize2;i++){
            for(j=0;j<csize;j++){
                printf("%d ",prod[i][j]);
            }
            printf("\n");
        }
    }
    else printf("Your given two matrices can't be multiplied as the number of columns in the first matrix and the number of rows in the second are not equal.");
    return 0;
}
