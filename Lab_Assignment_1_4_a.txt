#include <stdio.h>
#include <conio.h>
int two_sum(int arr1[],int,int,int pos[2]);
void main(){
    clrscr();
    int size,i,arr1[50],pos[2],sum,ch;
    printf("Enter the number of elements(<=50): ");
    scanf("%d",&size);
    for(i=0;i<size;i++){
        printf("Enter element %d: ",i+1);
        scanf("%d",&arr1[i]);
    }
    printf("Enter the target sum: ");
    scanf("%d",&sum);
    ch=two_sum(arr1,size,sum,pos);
    if(ch==1) printf("[%d,%d]",pos[0],pos[1]);
    else if(ch==0) printf("No two elements found with %d as their sum.",sum);
}
int two_sum(int arr1[],int size,int sum,int pos[2]){
    for(int i=0;i<size;i++){
        for(int j=i+1;j<size;j++){
            if(arr1[i]+arr1[j]==sum){
                pos[0]=i;
                pos[1]=j;
                return 1;
            }
        }
    }
return 0;
}
