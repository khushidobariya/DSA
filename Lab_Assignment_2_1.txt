#include <stdio.h>
#include <conio.h>
void main(){
int matrix[4][4],i,j;
for(i=0;i<=3;i++){
    for(j=0;j<=3;j++){
        if (i==j)
            matrix[i][j]=1;
        else
            matrix[i][j]=0;
    }
}
for(i=0;i<=3;i++){
    for(j=0;j<=3;j++){
        printf("%d ",matrix[i][j]);
    }
    printf("\n");
}
}
