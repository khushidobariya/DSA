#include <stdio.h>
#include <conio.h>
void main(){
int arr1[10],i,num,count=0;
for(i=0;i<=4;i++){
    printf("Enter element %d: ",i+1);
    scanf("%d",&arr1[i]);
}
printf("Enter the element to find in the array:");
scanf("%d",&num);
for(i=0;i<=4;i++){
    if(arr1[i]==num){
        printf("Found %d at position %d.\n",num,i+1);
        count+=1;
    }
}
if(count==0) printf("The given element not found in the array.");
}
